    <script>
        function calculateTotal() {
            const budget = parseFloat(document.getElementById('budgetInput').value) || 0;
            const isUrgent = document.getElementById('urgentSwitch').checked;
            const extra = isUrgent ? 20 : 0;
            const total = budget + extra;

            document.getElementById('totalDisplay').innerText = '¥ ' + total.toFixed(2);
        }

        function submitTask() {
            const budget = parseFloat(document.getElementById('budgetInput').value);
            if (!budget || budget <= 0) {
                alert("请输入有效的预算金额");
                return;
            }

            // 模拟提交 loading
            const btn = document.querySelector('.btn-primary');
            btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> 处理中...';
            btn.disabled = true;

            setTimeout(() => {
                // 关键跳转：带参数跳转到收银台，触发 Task 场景
                location.href = 'checkout.html?type=task';
            }, 1000);
        }
    </script>
